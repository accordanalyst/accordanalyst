anomaly_generic.pyPY
def flag(df, metric, w=6, z=2.5):
  base = df[metric].rolling(w).mean().shift(1)
  std  = df[metric].rolling(w).std().shift(1)
  return df[((df[metric]-base)/std).abs() >= z]
